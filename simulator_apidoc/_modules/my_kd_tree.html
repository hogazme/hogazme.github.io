class MyKdTree:
    def __init__(self, strict_naming=True):
        self.kdtree_arr = []
        self.kdtree_index = 0
        self.kdtree_naming_mapping = {}
        self.kdtree_naming_set = set()
        self.kdtree = None
        self.last_compiled_kdtree_index = 0
        self.strict_naming = strict_naming



[docs]
    def add_point(self, x, y, name):
        self.kdtree_naming_mapping[self.kdtree_index] = name
        self.kdtree_index += 1
        self.kdtree_arr.append([x, y])
        # check duplicate naming
        if name in self.kdtree_naming_set and self.strict_naming:
            print('Error: you feed duplicate name x=%f, y=%f, name=%s for a point' % (x, y, str(name)))
            # exit(1)
        self.kdtree_naming_set.add(name)

    def update_kd_tree(self):
        self.kdtree = cKDTree(np.array(self.kdtree_arr))
        self.last_compiled_kdtree_index = self.kdtree_index

    def dist_calculate(self, x1, y1, x2, y2):
        return ((x1 - x2) ** 2 + (y1 - y2) ** 2) ** (1 / 2)

    def find_nearest(self, x, y, limit=-1):
        if self.last_compiled_kdtree_index != self.kdtree_index:
            print('Rebuild kd-tree')
            self.update_kd_tree()

        results = []  # each element: ((x,y), name, kdtree-index, distance)
        if limit == -1:
            nearest_group = self.kdtree.query(np.array([x, y]), k=self.last_compiled_kdtree_index, workers=-1)[1].tolist()
        else:
            nearest_group = self.kdtree.query(np.array([x, y]), k=limit, workers=-1)[1].tolist()

        for candidate_index in nearest_group:
            coord = self.kdtree.data[candidate_index]
            distance = self.dist_calculate(x, y, coord[0], coord[1])
            results.append(((coord[0], coord[1]), self.kdtree_naming_mapping[candidate_index], candidate_index, distance))

        return results
