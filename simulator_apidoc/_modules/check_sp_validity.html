def check_single_sp(sp):
    """
    check if a single routing plan is valid in shp_G
    @param sp: shortest path
    @return:
    """
    node_set = set()
    try:
        for i in range(0, len(sp) - 1):
            if sp[i + 1] not in shp_G.nodes[sp[i]].next_Nodes.keys():
                print('err: not continuous')
                return False
            # else:
            #     print(shp_G.nodes[sp[i]].next_Nodes[sp[i + 1]].edge_length)
            if sp[i] not in node_set:
                node_set.add(sp[i])
            else:
                print('err: Node %d exists more than one time' % (sp[i]))
                return False
    except KeyError:
        print('err: node id not found')
    return True

for root, dirs, files in os.walk("./map_data/400k", topdown=False):
    for name in files:
        full_path = os.path.join(root, name)
        if '/route_info' in full_path and full_path.endswith('.csv'):
            print('check ' + full_path)
            with open(full_path, 'r') as file:
                sp = []
                time_seq = []
                rows = csv.DictReader(file)
                for row in rows:
                    sp.append(int(row['nodes_sequence']))
                    time_seq.append(float(row['nodes_timing']))
                if check_single_sp(sp):
                    bus_routes_lst.append(sp)
                    bus_seq_lst.append(time_seq)
                    bus_naming_lst.append(name.replace('.csv',''))
