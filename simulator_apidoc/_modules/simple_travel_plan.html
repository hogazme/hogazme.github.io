x -> bus stop 1: the passenger walks from x to bus stop 1

bus stop 2 -> y: the passenger walks from bus stop 2 to destination y

def get_nearest_node_id(center_node_id, type="bus_stop"):
    '''
    type: "bus_stop" or "point"
    '''
    nearest_group = shp_G.kdtree.query(np.array([shp_G.nodes[center_node_id].long, shp_G.nodes[center_node_id].lat]), k=len(shp_G.nodes), workers=-1)[
        1].tolist()
    top_k_nearest = []
    for nearest_node_id in nearest_group:
        if shp_G.nodes[nearest_node_id].node_type == type:  # NOTE: why not construct a subgraph with only bus stops?
            top_k_nearest.append(nearest_node_id)
        if len(top_k_nearest) > 10:
            break
    return top_k_nearest

def get_travel_plan(start_node_id, destination_node_id, transit_graph):
    """
    @shuocheng guo

    @param start_node_id: integer, shp_G node id as start node
    @param destination_node_id: integer, shp_G node id as destination node
    @return: you can use any format you want, so long as it contains such info: from node x to node y, you should walk/take bus (if take bus, please include which bus to take)
    """
    sub_graph = transit_graph.copy()

    # add edges to sub_graph
    # get nearest 10 bus stop to start_node_id
    nearest_bus_stops_from_start = get_nearest_node_id(start_node_id, type="bus_stop")
    nearest_bus_stops_from_destination = get_nearest_node_id(destination_node_id, type="bus_stop")

    # add edges to sub_graph
    for start in nearest_bus_stops_from_start:
        [sub_graph.add_edge(start_node_id, str(start) + "-" + str(rid), weight=300) for rid in
         range(len(all_bus_routes))]  # NOTE: hard code right now: 10 min walking time
    for end in nearest_bus_stops_from_destination:
        [sub_graph.add_edge(str(end) + "-" + str(rid), destination_node_id, weight=300) for rid in range(len(all_bus_routes))]

    # find the shortest path
    try:
        shortest_path = nx.shortest_path(sub_graph, start_node_id, destination_node_id, weight='weight')
        # print(f"shortest path: {shortest_path}")
    except nx.NetworkXNoPath:
        print('no path found')
        return None
    return parse_shortest_path(shortest_path)

def parse_shortest_path(shortest_path):
    '''
    (node_id, travel_mode, bus_route_id if applicable)
    travel_mode: "walk" or "bus"
    '''
    try:
        travel_plan = []
        travel_plan.append([shortest_path[0], "walk", None])
        travel_plan.append([int(shortest_path[1].split("-")[0]), "walk", None])
        for i in range(1, len(shortest_path) - 2):  # intermediate nodes
            if "-" in shortest_path[i]:
                # Note:
                # jiayi: I modify the integer bus index to formal name
                bus_route_index=int(shortest_path[i].split("-")[1])
                travel_plan.append([int(shortest_path[i].split("-")[0]), "bus", all_bus_namings[bus_route_index]])
            else:
                travel_plan.append([int(shortest_path[i]), "transfer", None])

        travel_plan.append([int(shortest_path[-2].split("-")[0]), "walk", None])
        travel_plan.append([shortest_path[-1], "walk", None])  # final node
    except IndexError:
        print('debug: %s' % (str(shortest_path)))

    return travel_plan

def set_up_transit_graph():
    transit_graph = nx.DiGraph()

    ## get transfer nodes, then add the transfer edges: node to transfer node, transfer node to node
    transfer_nodes = get_transfer_nodes(all_bus_routes)
    print('there are %d transfer nodes' % len(transfer_nodes))

    for rid, route in enumerate(all_bus_routes):
        for i in range(len(route) - 1):
            transit_graph.add_edge(str(route[i]) + "-" + str(rid), str(route[i + 1]) + "-" + str(rid),
                                   weight=max(all_bus_seq[rid][i + 1] - all_bus_seq[rid][i], 1))
            transit_graph.add_edge(str(route[i + 1]) + "-" + str(rid), str(route[i]) + "-" + str(rid),
                                   weight=max(all_bus_seq[rid][i + 1] - all_bus_seq[rid][i], 1))

            if route[i] in transfer_nodes:
                transit_graph.add_edge(str(route[i]), str(route[i]) + "-" + str(rid), weight=1)  # if 0, will get trapped.
                transit_graph.add_edge(str(route[i]) + "-" + str(rid), str(route[i]), weight=1)  # if 0, will get trapped.

    return transit_graph

def get_transfer_nodes(list_of_lists):
    from collections import Counter

    # Extend the list of lists to a single list
    long_list = [item for sublist in list_of_lists for item in sublist]

    # Count occurrences of each element
    element_counts = Counter(long_list)

    # Get elements that occur more than once
    transfer_nodes = [element for element, count in element_counts.items() if count > 1]

    return transfer_nodes
