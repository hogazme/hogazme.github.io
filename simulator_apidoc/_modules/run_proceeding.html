    def node_find_nearest_bus_stop_node(self, start_node_id):
        """
        Find the nearest bus stop and return bus stop's node it
        Will first try current zone using self.network
        If not found, then search the whole map
        Note:
        1. CANNOT find current node as nearest bus stop
        @return: bus stop node id
        """
        argmin_node_id=None
        # try to search in local kd-tree first
        start_t=time.time()
        for kdtree_query in self.local_kd_tree.find_nearest(shp_G.nodes[start_node_id].long,shp_G.nodes[start_node_id].lat):
            # based on format: ((x,y), name, kdtree-index, distance)
            # e.g.
            # ((6.0, 6.0), 'point (5,5)', 3, 0.7071067811865476)
            # so retrive ketree_query[1] -> name: in this case, name is the shp_G node id (NOT kdtree_index)
            candidate_node_id=kdtree_query[1]
            if shp_G.nodes[candidate_node_id].node_type=='bus_stop' and candidate_node_id!=start_node_id:
                argmin_node_id=candidate_node_id
                print('it took %f seconds to find nearest bus stop in local zone' % (time.time() - start_t))
                break

        if not argmin_node_id:
            start_t = time.time()
            # we cannot find local zone's neartest bus stop
            # use shp_G's kd-tree
            nearest_group = shp_G.kdtree.query(np.array([shp_G.nodes[start_node_id].long,shp_G.nodes[start_node_id].lat]), k=len(shp_G.nodes), workers=-1)[1].tolist()
            for nearest_node_id in nearest_group:
                if shp_G.nodes[nearest_node_id].node_type == 'bus_stop' and nearest_node_id!=start_node_id:
                    argmin_node_id = nearest_node_id
                    print('it took %f seconds to find nearest bus stop in global map' % (time.time() - start_t))
                    break
        return argmin_node_id

    def cross_boundary_node_id_generator(self, sp, curr_node, agent_belonging_rank, jump_first):
        """
        routing_queue_cross_process_index
        example:
        routing_queue: [10, 20, 100, 150, 200, 300, 400]
        Zone 1: 10, 20
        Zone 2: 100, 150
        Zone 3: 200, 300, 400
        Then routing_queue_cross_process_node_id should be [20, 150]
        ===== currently, routing_queue_cross_process_node_id's maximum load is 1 ======
        ===== ...Because we cannot get other processes' NodeAgent's property =====
        """
        if agent_belonging_rank != self.rank:
            print('agent_belonging_rank!=self.rank')
            exit(1)
        # the function of jump_first:
        # assuming a vehicle just transferred to another process, but it's still stopping at last process's node with node_id_mutex
        # jump_first prevents this function returning the same node id endlessly (in this case the vehicle will never move again)
        # jump_first=True in update() function
        # jump_first=False in init() and step() rerouting
        if not jump_first:
            sp_extend = [curr_node] + sp
        else:
            sp_extend = sp
        for index in range(1, len(sp_extend)):
            next_node_agent = self.node_agent_finder(sp_extend[index])
            if agent_belonging_rank != next_node_agent.uid[2]:
                return sp_extend[index - 1]

    def shortest_path_rustG(self, start_node_id, end_node_id):
        """
         shortest path function using RuestworkX

        Parameters:
            start_node_id: integer start node id (shp_G)
            end_node:id: integer end node id (shp_G)

        Return:
            sp: shortest path, list of integer shp_G node ids (remove heading)
            routing_queue_dist, shortest path distance, list of float values of shortest path's link length
        """

        path_mapping = rustworkx.digraph_dijkstra_shortest_paths(rust_G, start_node_id,
                                                                 end_node_id,
                                                                 float)
        if len(path_mapping) > 0:
            sp = list(path_mapping[next(iter(path_mapping))])
        else:
            sp = [start_node_id]

        routing_queue_dist = []
        if len(sp) >= 2:
            for i in range(0, len(sp) - 1):
                temp_dist = shp_G.nodes[sp[i]].next_Nodes[sp[i + 1]].edge_length
                routing_queue_dist.append(temp_dist)
        sp.pop(0)
        return sp, routing_queue_dist

    def adjust_dynamic_speed(self):
        """
        This function changes every vehicle's speed inside the zone
        current TOY FORMULAE:
        updated_speed = speed - 0.005 * (vehicle count)**2

        @return: None
        """

        # THE FOR-LOOP SHOULD RUN TWICE
        # FIRST TIME: update link load
        # SECOND TIME: update vehicle

        # iterate all vehicles within the zone and update the loading
        self.edge_loading = {}

        for v_id in agent_group:
            agent_i = agent_group[v_id]
            last_node_id = agent_i.routing_arrived[-1]
            next_node_id = agent_i.routing_queue[0]
            if next_node_id not in shp_G.nodes[last_node_id].next_Nodes.keys():
                print('error: dynamic speed: vehicle last node id is not connected to next node id')
                exit(1)
            link_key_str = str(last_node_id) + "->" + str(next_node_id)
            # update single tick variable
            if link_key_str not in self.edge_loading.keys():
                self.edge_loading[link_key_str] = 0
            self.edge_loading[link_key_str] += 1
            # update accumulate global variable as well
            if link_key_str not in self.accum_link_loading.keys():
                self.accum_link_loading[link_key_str] = 0
            self.accum_link_loading[link_key_str] += 1

        for v_id in agent_group:
            agent_i = agent_group[v_id]
            last_node_id = agent_i.routing_arrived[-1]
            next_node_id = agent_i.routing_queue[0]
            if next_node_id not in shp_G.nodes[last_node_id].next_Nodes.keys():
                print('adjust_dynamic_speed error: vehicle last node id is not connected to next node id')
                exit(1)
            link_key_str = str(last_node_id) + "->" + str(next_node_id)
            edge_length = shp_G.nodes[last_node_id].next_Nodes[next_node_id].edge_length
            link_loading = self.edge_loading[link_key_str]
            free_flow_vehicle_count = edge_length / 25
            if link_loading < free_flow_vehicle_count:
                adjust_alpha = 1 / 200
            else:
                adjust_alpha = 1 / 100
            agent_i.speed = max(agent_i.original_speed - adjust_alpha * (link_loading-1) ** 2, 1)

                    print("step() start_node_agent None error: %s, %s" % (vars(agent_i), node_agent_mapping))
                    exit(1)

                if shp_G.nodes[agent_i.node_id_mutex].node_type!='bus_stop':
                    # that means a vehicle just completed bus_stop -> home, and needs to find next job in nearest bus stop
                    end_node_id = self.node_find_nearest_bus_stop_node(agent_i.node_id_mutex)  # find nearest bus stop; this function guarantees to find one
                    if end_node_id == agent_i.node_id_mutex:
                        print('step() err: nearest bus stop is itself')
                        exit(1)
                else:
                    # that means a vehicle just completed last place -> bus stop, and needs to take the passenger to home
                    while True:
                        end_node_id = random.choice(shp_G.nodes).node_id
                        if end_node_id != agent_i.node_id_mutex:
                            break


                agent_i.destination = end_node_id
                sp, sp_dist = self.shortest_path_rustG(agent_i.node_id_mutex, end_node_id)
                agent_i.routing_queue = sp
                agent_i.routing_queue_dist = sp_dist
                agent_i.status = 1


                # 2024-02-15
                # debug something about vehicle: home->bus stop->home->bus stop->home...
                if shp_G.nodes[agent_i.node_id_mutex].node_type != 'bus_stop':
                    print('Vehicle %d is leaving node %d to nearest bus stop %d, distance %.2f meters' % (agent_i.agent_id, agent_i.node_id_mutex, end_node_id, sum(sp_dist)))
                else:
                    print("""Vehicle %d is leaving bus stop %d to passenger's home %d, distance %.2f meters""" % (agent_i.agent_id, agent_i.node_id_mutex, end_node_id, sum(sp_dist)))


                # change end_node_id for bus HERE:
                if agent_i.__class__.__name__ == 'BusAgent':
                    print('debug: a bus has arrived')
                    if agent_i.node_id_mutex == agent_i.start_station:
                        # currently the bus is located at the start_station
                        agent_i.destination = agent_i.end_station
                        agent_i.routing_queue = deepcopy(agent_i.bus_complete_sp)
                        agent_i.routing_queue_dist=deepcopy( agent_i.bus_complete_sp_dist)
                        agent_i.status = 1
                    elif agent_i.node_id_mutex == agent_i.end_station:
                        # currently the bus is located at the end_station
                        agent_i.destination = agent_i.start_station
                        sp_temp=deepcopy(agent_i.bus_complete_sp)
                        sp_dist_temp=deepcopy(agent_i.bus_complete_sp_dist)
                        sp_dist_temp.reverse()
                        sp_temp.pop(-1) # remove end station id
                        sp_temp.reverse()
                        sp_temp.append(agent_i.start_station)
                        # modify property
                        agent_i.routing_queue = deepcopy(sp_temp)
                        agent_i.routing_queue_dist = deepcopy(sp_dist_temp)
                        agent_i.status = 1
                    else:
                        print('err: bus cannot locate itself after arriving at start/stop station')
                        exit(1)
                # end bus route modify


                agent_i.routing_queue_cross_process_node_id = self.cross_boundary_node_id_generator(agent_i.routing_queue,
                                                                                                    agent_i.node_id_mutex,
                                                                                                    agent_i.belonging_rank,
                                                                                                    False)

                # a special case: the agent arrived at the boundary node and generate a new routing plan, but the new routing plan requires the vehicle:
                # ... to cross the boundary immediately
                if agent_i.routing_queue_cross_process_node_id == agent_i.node_id_mutex:
                    # print('debug: changing agent %d status to 2 in status==0 case'%(agent_id))
                    agent_i.status = 2

                agent_i.debug_history_routing_queue_cross_process_node_id_lst.append(
                    agent_i.routing_queue_cross_process_node_id)
                agent_i.debug_original_routing_queue = deepcopy(agent_i.routing_queue)

            if agent_i.status == 2:
                del_agent_id_lst.append(agent_id)
                local_node_agent = self.node_agent_finder(agent_i.node_id_mutex)
                target_node_agent = self.node_agent_finder(agent_i.routing_queue[0])
                if target_node_agent is None:
                    print('%s cannot find node %d in process %d, with node pool %s' % (
                        vars(agent_i), agent_i.routing_queue[0], self.rank, str(node_agent_mapping)))
                    exit(1)
                if target_node_agent.uid[2] == self.rank:
                    print(
                        'Why this vehicle %s is trying to transfer to current process %d' % (vars(agent_i), self.rank))
                    exit(1)
                # node_load_vehicle_dict structure:
                # key: local agent node reference
                # value: (agent_id, target agent node reference)
                if agent_i.node_id_mutex not in list(self.node_load_vehicle_dict.keys()):
                    self.node_load_vehicle_dict[agent_i.node_id_mutex] = [(agent_id, target_node_agent.uid[2])]
                else:
                    if agent_id not in self.node_load_vehicle_dict[agent_i.node_id_mutex]:
                        self.node_load_vehicle_dict[agent_i.node_id_mutex].append((agent_id, target_node_agent.uid[2]))

        # print('At %d, Rank %d, debug node_load_vehicle_dict: %s'%(self.runner.schedule.tick, self.rank,str(self.node_load_vehicle_dict)))
        self.context.synchronize(restore_agent)

        # delete after synchronize, due to the reason that a node can be re-visited multiple times
        # it's better to delete in batch, rather in save()
        for agent_id in del_agent_id_lst:
            del agent_group[agent_id]

        print(
            'After %d, process: %d, len(agent_group): %d, ' % (
                self.runner.schedule.tick + sys_proceeding_tick, self.rank, len(agent_group)))
        load_lst.append(len(agent_group))

        self.logging_coordinates()

def update_single_agent_next_tick_status(agent_i):
    """
    ========check cross boundary case first============
    Cross-boundary acts like a strong road bumper
    whatever the speed is, one vehicle has only two choices:
    1. Stop at the FIRST bumper(status=2)
    2. original logic
    """
    if agent_i.routing_queue_cross_process_node_id:
        # if agent_i.routing_queue_cross_process_node_id not in agent_i.routing_queue:
        #     print('%d not in %s, for agent %s'%(agent_i.routing_queue_cross_process_node_id, str(agent_i.routing_queue), vars(agent_i)))
        #     exit(1)
        if agent_i.routing_queue_cross_process_node_id == agent_i.node_id_mutex:
            # a special case: the vehicle is stopping by its cross-process-node (usually caused by spawning at cross-process-node)
            agent_i.routing_queue_cross_process_node_id = None  # waiting for next process's handle
            agent_i.status = 2
            # VERY IMPORTANT
            return
        else:
            routing_queue_cutting_index = agent_i.routing_queue.index(agent_i.routing_queue_cross_process_node_id)
            # print('agent %d have index %d'%(agent_i.agent_id, routing_queue_cutting_index))
            if agent_i.dist_compensation != 0:
                shadow_dist = agent_i.speed + agent_i.dist_compensation
            else:
                shadow_dist = agent_i.speed
            if shadow_dist >= sum(agent_i.routing_queue_dist[
                                  :routing_queue_cutting_index + 1]):  # this also applies to cutting index=-1's case

                # compensate distance loss after transferring
                # should do it now; prevent routing_queue_dist being changed later (del xxxxx)
                agent_i.dist_compensation = shadow_dist - sum(
                    agent_i.routing_queue_dist[:routing_queue_cutting_index + 1])

                # if agent_i.agent_id==0 and agent_i.dist_compensation!=0:
                #     print('At %d, agent 0 compensation %f'%(model.runner.schedule.tick, agent_i.dist_compensation ))
                #     # exit(1)

                agent_i.node_id_mutex = agent_i.routing_queue_cross_process_node_id
                agent_i.edge_id_mutex = None
                agent_i.long = shp_G.nodes[agent_i.node_id_mutex].long - shp_G.box_x
                agent_i.lat = shp_G.nodes[agent_i.node_id_mutex].lat - shp_G.box_y
                # debug
                agent_i.routing_arrived += agent_i.routing_queue[:routing_queue_cutting_index + 1]
                del agent_i.routing_queue[:routing_queue_cutting_index + 1]
                del agent_i.routing_queue_dist[
                    :routing_queue_cutting_index + 1]
                agent_i.routing_queue_cross_process_node_id = None  # waiting for next process's handle
                agent_i.status = 2

                # VERY IMPORTANT
                return

    # =========Original coordinates moving code===========
    # step 1: use routing_queue_dist to determine node operation
    # step 2: check destination arrival status first
    # step 3: update long and lat depending on #step2 arrival check
    # step 4: update agent status if necessary
    # step 5: don't forget to update routing_queue_dist
    arrived = False
    # if agent_i.agent_id == 0 and agent_i.dist_compensation != 0:
    #     print('At %d, agent 0 compensation %f' % (model.runner.schedule.tick, agent_i.dist_compensation))
    #     exit(1)

    if agent_i.dist_compensation != 0:
        shadow_dist = agent_i.speed + agent_i.dist_compensation
        agent_i.dist_compensation = 0
    else:
        shadow_dist = agent_i.speed

    if not agent_i.routing_queue_dist:
        arrived = True
    elif shadow_dist < agent_i.routing_queue_dist[0]:
        arrived = False
    elif shadow_dist >= sum(agent_i.routing_queue_dist):
        arrived = True
    if arrived:
        # just arrived
        """
        The following code:
        once arrived, set status to 0 and clear all targets/movements
        """
        # first: update the location as destination
        agent_i.routing_arrived += agent_i.routing_queue
        agent_i.node_id_mutex = agent_i.destination
        agent_i.edge_id_mutex = None
        agent_i.long = shp_G.nodes[agent_i.node_id_mutex].long - shp_G.box_x
        agent_i.lat = shp_G.nodes[agent_i.node_id_mutex].lat - shp_G.box_y
        # Second: clear and wait for re-routing
        agent_i.routing_queue_dist = []
        agent_i.routing_queue = []
        agent_i.status = 0
        # Extra
        agent_i.routing_queue_cross_process_node_id = None

    else:
        pop_index = 0
        while shadow_dist - agent_i.routing_queue_dist[pop_index] >= 0:
            shadow_dist -= agent_i.routing_queue_dist[pop_index]
            pop_index += 1
        # status: just cannot arrive
        if pop_index > 0:
            agent_i.routing_arrived.extend(agent_i.routing_queue[:pop_index])
            del agent_i.routing_queue_dist[:pop_index]
            del agent_i.routing_queue[:pop_index]
        agent_i.routing_queue_dist[0] -= shadow_dist  # NO LENGTH CHECK, JUST CONFIDENT IN MY **** CODE
        # From node self.routing_arrived[-1] to node [self.next_node]
        # distance to self.routing_arrived[-1]: shadow_dist
        start_node = shp_G.nodes[agent_i.routing_arrived[-1]]
        end_node = shp_G.nodes[agent_i.routing_queue[0]]
        # print("%s, start=%d, end=%d" %(str(self.routing_queue), start_node.node_id, end_node.node_id))
        if pop_index > 0:
            agent_i.long = start_node.long + shadow_dist * \
                           start_node.next_Nodes[end_node.node_id].unit_vector[
                               0] - shp_G.box_x
            agent_i.lat = start_node.lat + shadow_dist * start_node.next_Nodes[end_node.node_id].unit_vector[
                1] - shp_G.box_y
        else:
            agent_i.long = agent_i.long + shadow_dist * start_node.next_Nodes[end_node.node_id].unit_vector[
                0]  # does not need to minus box base coord
            agent_i.lat = agent_i.lat + shadow_dist * start_node.next_Nodes[end_node.node_id].unit_vector[
                1]  # does not need to minus box base coord
