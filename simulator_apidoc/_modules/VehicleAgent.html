    def assign_attributes(self, agent_id, node_id_mutex, edge_id_mutex, long, lat, speed, destination, status, routing_queue,
                          routing_queue_dist, routing_queue_cross_process_node_id, belonging_rank, capacity, loading):
        """
        Due to code design, a vehicle agent's attributes will be assigned HERE, using this function

        @param agent_id: the vehicle's universe ID (rather than process's local ID)
        @param node_id_mutex: the vehicle's spawn shp_G node id (int)
        @param edge_id_mutex: (this value is actually not used)
        @param long: the vehicle's spawn longitude (UTM, float)
        @param lat: the vehicle's spawn latitude (UTM, float)
        @param speed: the vehicle's speed (float)
        @param destination:  the vehicle's destination (shp_G node id, integer)
        @param status: the vehicle's status (integer), currently we use 0 as "idle", 1 as "more", 2 as "waiting at the boundary to be transferred)
        @param routing_queue: the vehicle's current shortest path routing queue (list of integers representing shp_G node ids)
        @param routing_queue_dist: the vehicle's current shortest path break-down link length (list of floats representing each link's length)
        @param routing_queue_cross_process_node_id: updated at: 1. init 2. re-partitioning, to decide which node is the decisive boundary node to wait and be transferred
        @param belonging_rank: integer, the vehicle's belonging rank
        @return:
        """

        # mutex variable pair:
        # node_id & edge_id
        self.agent_id = agent_id
        self.node_id_mutex = node_id_mutex
        self.edge_id_mutex = edge_id_mutex
        self.long = long
        self.lat = lat
        self.speed = speed
        self.original_speed = speed
        self.destination = destination
        self.status = status  # status: 1: moving towards destination; 0: idle
        # navigation
        # next routing nodes, using node_id format
        self.routing_queue = routing_queue
        # next routing nodes' distance(s), strictly corresponding to self.routing_queue
        self.routing_queue_dist = routing_queue_dist
        self.routing_queue_cross_process_node_id = routing_queue_cross_process_node_id
        self.belonging_rank = belonging_rank
        self.capacity=capacity
        self.loading=loading

        # these variables do not need outside attributes assign
        self.routing_arrived = [node_id_mutex]
        self.dist_compensation = 0
        self.loaded_human=[]
        self.total_served_passenger=0

        # special copy, for debugging only

        self.debug_original_routing_queue = deepcopy(self.routing_queue)
        self.debug_original_spawn = self.node_id_mutex
        self.debug_history_routing_queue_cross_process_node_id_lst = [routing_queue_cross_process_node_id]
        self.debug_routing_arrived_nodeAgent = deepcopy(self.routing_arrived)
        self.debug_belong_rank_history = [self.belonging_rank]

class BusAgent(VehicleAgent):
    def __init__(self, vehicleAgent_obj, capacity, loading, start_station, end_station, bus_complete_sp, bus_complete_sp_dist, bus_stop_route,
                 bus_stop_route_schedule, bus_route_naming):
        """

        @param vehicleAgent_obj: the vehicle object reference
        @param capacity: integer, the maximum capacity
        @param loading:  integer, the bus's current loaded number of people (including driver)
        @param start_station: the bus's start station (integer, shp_G node id)
        @param end_station: the bus's end station (integer, shp_G node id)
        @param bus_complete_sp: the bus's current shortest path routing queue (list of integers representing shp_G node ids)
        @param bus_complete_sp_dist: the bus's current shortest path break-down link length (list of floats representing each link's length)
        @param bus_stop_route: list of integer shp_G node ids, not necessarily to be continuous
        @param bus_stop_route_schedule:  the strict one-to-one mapping for bus_stop_route list. example: [0, 360, 720, 1000] means the bus should arrive node 49 at tick 0
        """

        super().__init__()
        copied_vars = deepcopy(vars(vehicleAgent_obj))
        # remove these attributes: generated not using function parameter (instead, generated inside the function)
        copied_vars.pop('routing_arrived', None)
        copied_vars.pop('dist_compensation', None)
        copied_vars.pop('original_speed', None)
        copied_vars.pop('loaded_human', None)
        copied_vars.pop('total_served_passenger', None)

        copied_vars.pop('debug_original_routing_queue', None)
        copied_vars.pop('debug_original_spawn', None)
        copied_vars.pop('debug_history_routing_queue_cross_process_node_id_lst', None)
        copied_vars.pop('debug_routing_arrived_nodeAgent', None)
        copied_vars.pop('debug_belong_rank_history', None)
        # end
        self.assign_attributes(**copied_vars)
        self.capacity = capacity
        self.loading = loading
        self.start_station = start_station
        self.end_station = end_station
        self.bus_complete_sp = bus_complete_sp
        self.bus_complete_sp_dist = bus_complete_sp_dist
        self.bus_stop_route = bus_stop_route  # the discrete bus stop node id list; pop up the first element as the bus's "temp destination" until no element exists
        self.bus_stop_route_schedule = bus_stop_route_schedule
        self.bus_route_naming=bus_route_naming
        # dynamic variable, minus 1 for every tick, replenish to 10 once arriving at each bus stop. if this variable equals to 0, we minus 1 again (to -1) and start the bus.
        # -1: moving
        # 10: arrived at a bus stop
        # 0~10: waiting for passengers
        # 0: set to -1 and next tick move again
        self.bus_stop_countdown = -1
