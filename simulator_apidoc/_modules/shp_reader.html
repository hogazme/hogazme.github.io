def run():
    shp_G = graph.Graph()
    with open('./map_data/400k/nodes.csv', 'r') as file:
        rows = csv.DictReader(file)
        for row in rows:
            x = float(row['x'])
            y = float(row['y'])
            node_type = row['type']
            shp_G.add_node(x, y, node_type)

    with open('./map_data/400k/edges.csv', 'r') as file:
        rows = csv.DictReader(file)
        for row in rows:
            x1 = float(row['x1'])
            y1 = float(row['y1'])
            x2 = float(row['x2'])
            y2 = float(row['y2'])
            shp_G.add_edge(x1, y1, x2, y2)

    # now check if all edges are two-way directed
    # and check node id as well
    index = 0
    for node in shp_G.nodes:
        if node.node_id != index:
            print('error: node id not strictly assigned')
            exit(1)
        index += 1
        for next_node_id in node.next_Nodes.keys():
            if node.node_id not in shp_G.nodes[next_node_id].next_Nodes.keys():
                print('error: not all edges are two-way directed')
                exit(1)

    # 2024-02-15
    # build kd-tree
    kdtree_arr = []
    check_index = 0
    for node in shp_G.nodes:
        if node.node_id != check_index:
            print('err when building shp_G kd-tree')
            exit(1)
        kdtree_arr.append([node.long, node.lat])
        check_index += 1

    kdtree_points = np.array(kdtree_arr)
    shp_G.kdtree = cKDTree(kdtree_points)

    return shp_G




if __name__ == '__main__':
    run()
