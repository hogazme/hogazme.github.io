def run():
    # read the shapefile into the rustworkx graph
    # ensure that each edge is added twice, as the directed graph
    shp_G = shp_reader.run()
    rust_node_n = len(shp_G.nodes)
    rust_g = rx.PyDiGraph()
    for i in range(0, len(shp_G.nodes)):
        rust_g.add_node(i)
    for node in shp_G.nodes:
        for neighbor_node_id in node.next_Nodes:
            rust_g.add_edge(node.node_id, neighbor_node_id, node.next_Nodes[neighbor_node_id].edge_length)
            # rust_g.add_edge(neighbor_node_id, node.node_id, shp_reader.node_distance(node, shp_G.nodes[neighbor_node_id]))

    # test the node to itself!
    print('-----begin rustworkx validity test-----')
    test_node = random.choice(shp_G.nodes)
    # trace to itself, len(path)==1
    path_mapping = rx.digraph_dijkstra_shortest_paths(rust_g, test_node.node_id, test_node.node_id,
                                                      float)
    if len(path_mapping) > 0:
        print(list(path_mapping[next(iter(path_mapping))]))
    else:
        print('[]')
    # trace to it's random neighbor, len(path)==2
    rand_neighbor_id = random.choice(list(test_node.next_Nodes.keys()))
    path_mapping = rx.digraph_dijkstra_shortest_paths(rust_g, test_node.node_id,
                                                      rand_neighbor_id,
                                                      float)

    # debug2 = list(path_mapping[next(iter(path_mapping))])

    if len(path_mapping) > 0:
        print(list(path_mapping[next(iter(path_mapping))]))
    else:
        print('[]')
    # len(path) should be 3 or 1, depending on GOD RANDOM
    # len=3: trace to random neighbor's random neighbor
    # len=1: trace to itself
    rand_neighbor_neighbor_id = random.choice(list(shp_G.nodes[rand_neighbor_id].next_Nodes.keys()))
    path_mapping = rx.digraph_dijkstra_shortest_paths(rust_g, test_node.node_id, rand_neighbor_neighbor_id,
                                                      float)
    if len(path_mapping) > 0:
        print(list(path_mapping[next(iter(path_mapping))]))
    else:
        print('[]')

    print('-----end graph validity test-----')

    return shp_G, rust_g
