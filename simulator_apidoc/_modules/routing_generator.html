def shortest_path_rustG(start_node_id, end_node_id):
    path_mapping = rustworkx.digraph_dijkstra_shortest_paths(rust_G, start_node_id,
                                                             end_node_id,
                                                             float)
    if len(path_mapping) > 0:
        sp = list(path_mapping[next(iter(path_mapping))])
    else:
        sp = [start_node_id]

    routing_queue_dist = []
    if len(sp) >= 2:
        for i in range(0, len(sp) - 1):
            temp_dist = shp_G.nodes[sp[i]].next_Nodes[sp[i + 1]].edge_length
            routing_queue_dist.append(temp_dist)
    sp.pop(0)
    return sp, routing_queue_dist

def find_proper_bus_route(origin_node_idx):
    origin_node = shp_G.nodes[origin_node_idx]

    nearest_group = \
        kdtree.query(np.array([origin_node.long, origin_node.lat]), k=len(shp_G.nodes), workers=-1)[1].tolist()
    # print('origin: %s' % (print_node_info_by_index(origin_node_idx)))
    for candidate_idx in nearest_group:
        dist = dist_calculate(x1=origin_node.long, y1=origin_node.lat, x2=shp_G.nodes[candidate_idx].long,
                              y2=shp_G.nodes[candidate_idx].lat)
        # print('idx: %d, dist: %f' % (idx, dist))
        if dist >= 1.5 * 1.6 * 1000 :
            # at least 0.8 mile
            # print('find dist from Node %d (%f, %f) to Node %d (%f, %f)'%(origin_node_idx, origin_node.long, origin_node.lat, candidate_idx,shp_G.nodes[candidate_idx].long, shp_G.nodes[candidate_idx].lat ))
            return candidate_idx

    return None
